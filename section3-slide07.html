GEMINI AI WITH FLASK

Flask integrates Gemini imagen-3.0 to generate images. The model receives prompts and returns images in base64 format. 

import google.generativeai as genai
import os

genai.configure(api_key=os.getenv('GEMINI_API_KEY'))

def generate_pokemon_image(prompt):
    model = genai.GenerativeModel('imagen-3.0-generate-001')
    
    result = model.generate_images(
        prompt=f"Pokemon style: {prompt}",
        number_of_images=1,
        aspect_ratio="1:1",
        safety_filter_level="block_only_high"
    )
    
    # Convertir imagen a base64
    image_data = result.images[0]
    base64_image = image_data._image_bytes
    
    return base64_image
