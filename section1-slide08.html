Files with [parameter] create dynamic routes. Perfect for detail pages like /pokemon/pikachu. 

---
// pages/pokemon/[id].astro

export async function getStaticPaths() {
    // Generate all routes at build time
    const response = await fetch('https://pokeapi.co/api/v2/pokemon?limit=151');
    const data = await response.json();
    
    return data.results.map((pokemon) => ({
        params: { id: pokemon.name },
        props: { url: pokemon.url }
    }));
}

const { id } = Astro.params;
const { url } = Astro.props;

const pokemonData = await fetch(url).then(r => r.json());

<Layout title={`Pokémon: ${id}`}>
